package org.robotlegs.examples.ModularDoodads.modules.doodad
{
    import flash.display.DisplayObjectContainer;
    
    import org.robotlegs.core.IInjector;
    import org.robotlegs.utilities.modular.mvcs.ModuleContext;
    
    public class DoodadModuleContext extends ModuleContext
    {
        public function DoodadModuleContext(contextView:DisplayObjectContainer, parentInjector:IInjector)
        {
            super(contextView, true, parentInjector);
        }
        
        override public function startup():void
        {
            mediatorMap.mapView(DoodadModule, DoodadModuleMediator);
        }
    }
}
