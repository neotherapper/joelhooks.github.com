package
{
    import com.bit101.components.PushButton;
    
    import flash.display.Sprite;
    import flash.display.StageAlign;
    import flash.display.StageScaleMode;
    import flash.events.Event;
    import flash.geom.Point;
    
    import org.robotlegs.examples.ModularDoodads.ModularDoodadsContext;
    import org.robotlegs.examples.ModularDoodads.common.events.DoodadModuleEvent;
    import org.robotlegs.examples.ModularDoodads.common.events.LoggingEvent;
    import org.robotlegs.examples.ModularDoodads.modules.doodad.DoodadModule;
    import org.robotlegs.examples.ModularDoodads.modules.logger.LoggerModule;
    import org.robotlegs.utilities.modular.core.IModuleContext;
    
    [SWF("450", height="550", frameRate="30", backgroundColor="#FFFFFF")]
    public class ModularDoodads extends Sprite
    {
        private var context:IModuleContext;
        
        private var button:PushButton = new PushButton(Sprite(this),0,0,"Create", handleLogIt);
        private var colorButton:PushButton = new PushButton(Sprite(this),110,5,"Colors", handleFlash);
        
        public function ModularDoodads()
        {
            stage.align = StageAlign.TOP_LEFT;
            stage.scaleMode = StageScaleMode.NO_SCALE;
            context = new ModularDoodadsContext(this);
            addChild(button);
            button.y = 5;
        }
        
        public function createLoggerView():void
        {
            var logger:LoggerModule = new LoggerModule();
            logger.y = 500;
            addChild(logger); 
        }
        
        private function handleLogIt(event:Event):void
        {
            var doodad:DoodadModule = new DoodadModule(Sprite(this));
            dispatchEvent(new LoggingEvent(LoggingEvent.LOG_MESSAGE, "add a doodad..."));
        }
        
        private function handleFlash(event:Event):void
        {
            dispatchEvent(new DoodadModuleEvent(DoodadModuleEvent.CHANGE_DOODAD_COLORS));
            dispatchEvent(new LoggingEvent(LoggingEvent.LOG_MESSAGE, "Change your colors..."));
        }
        
        public function resetFocus():void
        {
            stage.focus = button;
        }
    }
}
