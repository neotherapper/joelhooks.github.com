package flexlib.mdi.containers
{
    /**
     * Enumeration of possible window states.
     */
    public class MDIWindowState
    {
        /**
         * Window is floating.
         */
        public static const NORMAL:int = 0;
        /**
         * Window is minimized.
         */
        public static const MINIMIZED:int = 1;
        /**
         * Window is maximized.
         */
        public static const MAXIMIZED:int = 2;
    }
}
