package org.robotlegs.examples.ModularDoodads.modules.doodad
{
    import com.bit101.components.PushButton;
    import com.bit101.components.VBox;
    import com.bit101.components.Window;
    
    import flash.display.DisplayObjectContainer;
    import flash.display.Sprite;
    import flash.events.Event;
    import flash.events.TimerEvent;
    import flash.geom.Point;
    import flash.utils.Timer;
    
    import org.robotlegs.core.IInjector;
    import org.robotlegs.examples.ModularDoodads.common.events.DoodadModuleEvent;
    import org.robotlegs.utilities.modular.core.IModule;
    import org.robotlegs.utilities.modular.core.IModuleContext;
    
    public class DoodadModule extends Window implements IModule
    {
        private var context:IModuleContext;
        private var background:Sprite;
        private var doStuff:PushButton
        private var remove:PushButton 
        private var buttonContainer:VBox ;
        private var flashTimer:Timer
        private var colorCache:uint;
        private var currentColor:uint;
        
        public function set backgroundColor(value:uint):void
        {
            currentColor = value*Math.random();
            background.graphics.clear();
            background.graphics.beginFill(currentColor);
            background.graphics.drawRect(0,0,100,80);
        }
        
        public function DoodadModule(parent:DisplayObjectContainer)
        {
            super(parent, 50, 50, "Doodad");
            initChildren();
        }

        private function initChildren():void
        {
            flashTimer = new Timer(50,4);
        
            background = new Sprite();
            backgroundColor = 0xFFFFFF;
            content.addChild(background);
        
            buttonContainer = new VBox(content);            
            buttonContainer.width = 80;
            buttonContainer.spacing = 5;
            buttonContainer.height = 45;
            buttonContainer.x = 10;
            buttonContainer.y = 18;
        
            doStuff = new PushButton(buttonContainer, 0, 0, "flash em", handleDoStuff); 
            doStuff.width = 80;
            remove = new PushButton(buttonContainer, 0, 0, "remove me", handleRemoveMe); 
            remove.width = 80;
        }

        
        [Inject]public function set parentInjector(value:IInjector):void
        {
            context = new DoodadModuleContext(this, value);
        }
        
        public function dispose():void
        {
            context.dispose();
            context = null;
            flashTimer = null;
            while(buttonContainer.numChildren > 0)
                buttonContainer.removeChildAt(0);
            while(content.numChildren > 0)
                content.removeChildAt(0);
            remove = null;
            doStuff = null;
            background = null;
            buttonContainer = null;
        }
        
        private function handleDoStuff(event:Event):void
        {
            dispatchEvent(new DoodadModuleEvent(DoodadModuleEvent.FLASH_DOODADS));
        }
        
        private function handleRemoveMe(event:Event):void
        {
            dispatchEvent(new DoodadModuleEvent(DoodadModuleEvent.REMOVE_DOODAD));
        }
        
        public function flash():void
        {
            flashTimer.reset();
            colorCache = currentColor;
            flashTimer.addEventListener(TimerEvent.TIMER, handleFlashTimer, false, 0, true);
            flashTimer.addEventListener(TimerEvent.TIMER_COMPLETE, handleFlashTimerComplete, false, 0, true);
            flashTimer.start();
        }
        
        public function removeMe():void
        {
            stage.focus = null;
            if(parent && parent.contains(this))
                parent.removeChild(this);
            dispose();
        }
        
        private function handleFlashTimer(event:TimerEvent):void
        {
            if(currentColor == colorCache)
                backgroundColor = 0xFFFFFF;
            else
                backgroundColor = colorCache;
        }
        
        private function handleFlashTimerComplete(event:TimerEvent):void
        {
            flashTimer.removeEventListener(TimerEvent.TIMER, handleFlashTimer);
            flashTimer.removeEventListener(TimerEvent.TIMER_COMPLETE, handleFlashTimerComplete);    
            backgroundColor = colorCache;
        }
    }
}
