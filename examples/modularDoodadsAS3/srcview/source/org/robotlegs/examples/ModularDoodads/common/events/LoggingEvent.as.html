package org.robotlegs.examples.ModularDoodads.common.events
{
    import flash.events.Event;
    
    public class LoggingEvent extends Event
    {
        public static const LOG_MESSAGE:String = "logMessage";
        
        private var _message:String;

        public function get message():String
        {
            return _message;
        }
        
        public function LoggingEvent(type:String, message:String, bubbles:Boolean=false, cancelable:Boolean=false)
        {
            super(type, bubbles, cancelable);
            _message = message;
        }
        
        override public function clone():Event
        {
            return new LoggingEvent(type,message,bubbles,cancelable);
        }
    }
}
