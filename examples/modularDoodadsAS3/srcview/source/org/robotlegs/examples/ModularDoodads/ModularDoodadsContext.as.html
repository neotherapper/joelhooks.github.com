package org.robotlegs.examples.ModularDoodads
{
    import flash.display.DisplayObjectContainer;
    
    import org.robotlegs.core.IInjector;
    import org.robotlegs.examples.ModularDoodads.modules.doodad.DoodadModule;
    import org.robotlegs.examples.ModularDoodads.modules.logger.LoggerModule;
    import org.robotlegs.examples.ModularDoodads.view.ModularDoodadsMediator;
    import org.robotlegs.utilities.modular.mvcs.ModuleContext;
    
    public class ModularDoodadsContext extends ModuleContext
    {
        public function ModularDoodadsContext(contextView:DisplayObjectContainer)
        {
            super(contextView);
        }
        
        override public function startup():void
        {
            viewMap.mapType(LoggerModule);
            viewMap.mapType(DoodadModule);
            mediatorMap.mapView(ModularDoodads, ModularDoodadsMediator);
        }
    }
}
