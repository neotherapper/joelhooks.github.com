    clipContent="true" 
    borderStyle="solid"

     <mx:Metadata>
         [Event(name="mouseOverItem", type="com.joelhooks.containers.pleatClasses.PleatEvent")]
         [Event(name="mouseOutItem", type="com.joelhooks.containers.pleatClasses.PleatEvent")]

     <mx:Script>
         <![CDATA[
             import com.joelhooks.containers.pleatClasses.controls.PleatControls;
             import mx.core.IContainer;
             import com.joelhooks.containers.Pleat;
             import mx.effects.easing.Quadratic;
             import com.gskinner.motion.GTween;
             
             private static const TWEEN_DELAY:Number = .1;
             
             [Bindable] public var labelText:String;
             [Bindable] public var index:int;
             [Bindable] public var selected:Boolean = false;
             [Bindable] private var pleatItemTotal:int = 0;
             [Bindable] public var iconClass:Class;
             
             [Inspectable(arrayType="DisplayObject")]
             [Bindable] private var _controls:Array;
             
             public function get controls():Array
             {
                 return this._controls;
             }
             
             public function set controls(value:Array):void
             {
                 this._controls = value;
             }
             
             override protected function createChildren():void
             {
                 super.createChildren();
                 for each(var displayObject:DisplayObject in this.controls)
                 {
                     this.controlContainer.addChild(displayObject);
                 }
             }
             
             public var openWidth:Number;
             
             private var open:Boolean = false;
             
             private function init():void
             {
                 this.pleatItemTotal = this.pleat.getPleatItemCount();
                if(this.index==0)
                {
                    this.handleClick();
                }
                
             }
             
             private function handleMouseOver(event:MouseEvent):void
             {
                 if(this.open)
                     return;
                 var timer:Timer = new Timer(750);
                 timer.addEventListener(TimerEvent.TIMER, mouseOverTimerComplteHandler)
                timer.start();
                 //this.dispatchEvent(new PleatEvent(PleatEvent.MOUSE_OVER_ITEM));
                 event.updateAfterEvent();
                 this.mouseBox.addEventListener(Event.ENTER_FRAME, handleOverCheck)
             }
             
             private function handleOverCheck(event:Event):void
             {
                 if(!this.mouseIsOver)
                 {
                     this.mouseBox.removeEventListener(Event.ENTER_FRAME, handleOverCheck)
                     this.dispatchEvent(new PleatEvent(PleatEvent.MOUSE_OUT_ITEM));
                 }
             }
             
             private function mouseOverTimerComplteHandler(event:TimerEvent):void
             {
                 event.target.stop()
                 event.target.removeEventListener(TimerEvent.TIMER, mouseOverTimerComplteHandler);
                 trace(this.labelText,this.mouseBox.mouseX,this.mouseBox.mouseY)    
                 if(this.mouseIsOver) 
                     this.dispatchEvent(new PleatEvent(PleatEvent.MOUSE_OVER_ITEM));
                 else
                     trace("MOUSE IS OUT");            
             }
             
             private function handleMouseOut(event:MouseEvent):void
             {
                 if(!this.mouseIsOver) 
                     this.dispatchEvent(new PleatEvent(PleatEvent.MOUSE_OUT_ITEM));
             }
             
             private function get mouseIsOver():Boolean
             {
                 return (this.mouseX > 0 && this.mouseX < this.width && this.mouseY > 0 && this.mouseY < this.height) ;
             }
             
             private function handleClick():void
             {
                 this.dispatchEvent(new PleatEvent(PleatEvent.SELECT_ITEM));
             }
             
             public function close(offset:int=0):void
             {
                 if(this.selected&&!open)
                     return
                 var tween2:GTween = new GTween(this,TWEEN_DELAY,{x:(this.pleatIcon.width*(this.index))+offset});
                 var tween:GTween = new GTween(this,TWEEN_DELAY,{width:this.pleatIcon.width});
                 this.open = false;
                 trace("CLOSE ME",index)
                 this.mouseBox.includeInLayout = this.mouseBox.visible = true
             }
             
             public function reposition(offset:int=0):void
             {
                 var tween2:GTween = new GTween(this,TWEEN_DELAY,{x:(this.pleatIcon.width*(this.index))+offset});
             }
             
             public function openLabel(direction:String):void
             {
                 /* if(this.open)
                     return;
                     this.open = true; */
                 var tween:GTween;
                 if(direction=="left" )
                 {
                     tween = new GTween(this,TWEEN_DELAY,{width:this.pleatIcon.width+this.pleatLabel.width,x:pleat.width-(this.pleatIcon.width*(pleat.getPleatItemCount()-index))-this.pleatLabel.width},{completeListener:openComplete});
                 }
                 else
                 {
                     tween = new GTween(this,TWEEN_DELAY,{width:this.pleatIcon.width+this.pleatLabel.width},{completeListener:openComplete});
                 }
                 trace("OPEN ME LABEL",index)
             }
             
             private function openComplete(obj:Object):void
             {
                 //this.open = true;
             }
             
             public function closeLabel(direction:String):void
             {
                 /* if(!this.open||this.selected)
                     return; */
                 var tween:GTween;
                 this.open = false
                 if(direction=="left" )
                 {
                     tween = new GTween(this,TWEEN_DELAY,{width:this.pleatIcon.width,x:pleat.width-(this.pleatIcon.width*(pleat.getPleatItemCount()-index))},{completeListener:closeComplete});
                 }
                 else
                 {
                     tween = new GTween(this,TWEEN_DELAY,{width:this.pleatIcon.width},{completeListener:closeComplete});
                 }
                 trace("CLOSE ME LABEL",index)
             }
             
             private function closeComplete(obj:Object):void
             {
                 //this.open = false;
             }
             
             public function openContent():void
             {
                 this.open = true;
                 var tween:GTween = new GTween(this,TWEEN_DELAY,{width:this.pleatIcon.width+this.pleatLabel.width+this.pleatContent.width});
                 var tween2:GTween = new GTween(this,TWEEN_DELAY,{x:(this.pleatIcon.width*(this.index))});
                 this.mouseBox.includeInLayout = this.mouseBox.visible = false;
             }
             
             public function resizeContent(amount:Number=0):void
             {
                 var tween:GTween = new GTween(this,TWEEN_DELAY,{width:this.pleatIcon.width+this.pleatLabel.width+this.pleatContent.width-amount});
             }
             
             public function calculateOpenWidth():void
             {
                 this.validateNow();
                 this.openWidth = this.pleatContent.width+ this.pleatLabel.width+this.pleatIcon.width
                 
                 trace(openWidth)
             }
             
             [Bindable] protected function get pleat():Pleat
             {
                 return this.parent as Pleat;
             }
             
             protected function set pleat(value:Pleat):void
             {
                 
             }
