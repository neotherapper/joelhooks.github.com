package org.robotlegs.examples.ModularDoodads.modules.logger
{
    import org.robotlegs.examples.ModularDoodads.common.events.LoggingEvent;
    import org.robotlegs.utilities.modular.mvcs.ModuleMediator;
    
    public class LoggerModuleMediator extends ModuleMediator
    {
        [Inject]public var view:LoggerModule;
        
        override public function onRegister():void
        {
            addModuleListener(LoggingEvent.LOG_MESSAGE, handleLogMessage, LoggingEvent);
        }
        
        private function handleLogMessage(event:LoggingEvent):void
        {
            view.addLoggingMessage(event.message);
        }
    }
}
