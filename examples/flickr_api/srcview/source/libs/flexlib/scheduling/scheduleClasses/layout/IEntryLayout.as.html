    import mx.events.CollectionEvent;

    public interface IEntryLayout extends ILayoutProvider
    {

        function get rowHeight() : Number;
        function set rowHeight( value : Number ) : void;

        function createLayout() : void;

        function removeItem( event : CollectionEvent ) : void;

        function updateItem( event : CollectionEvent ) : void;
        function resetItem( event : CollectionEvent ) : void;
        function refreshItem( event : CollectionEvent ) : void;
        function moveItem( event : CollectionEvent ) : void;
    }
}
