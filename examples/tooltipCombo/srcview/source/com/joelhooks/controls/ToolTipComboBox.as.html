package com.joelhooks.controls
{
    import flash.events.Event;
    
    import mx.controls.ComboBox;

    public class ToolTipComboBox extends ComboBox
    {
        public function ToolTipComboBox()
        {
            super();
        }
        
        //----------------------------------
        //  toolTipFunction
        //----------------------------------
    
        /**
         *  @private
         *  Storage for the toolTipFunction property.
         */
        private var _toolTipFunction:Function = null;
    
        [Bindable("toolTipFunctionChanged")]
        [Inspectable(category="Other")]
        /**
         *  User-supplied function to run on the <code>selectedItem</code> to determine its tooltip..
         * 
         *  <p>The toolTipFunction takes a single argument which is the item
         *  in the dataProvider and returns a String:</p>
         *  <pre>
         *  myToolTipFunction(item:Object):String
         *  </pre>
         *
         * @return 
         * 
         */    
        public function get toolTipFunction():Function
        {
            return _toolTipFunction;
        }
    
        /**
         *  @private
         */
        public function set toolTipFunction(value:Function):void
        {
            _toolTipFunction = value;
            this.invalidateDisplayList();
            dispatchEvent(new Event("toolTipFunctionChanged"));
        }    
        
        override protected function updateDisplayList(unscaledWidth:Number,
                                                  unscaledHeight:Number):void
        {
            super.updateDisplayList(unscaledWidth, unscaledHeight)
            
            if(this._toolTipFunction!=null)
            {
                this.toolTip = this._toolTipFunction(this.selectedItem);
            
            }            
        }    
    }
}
