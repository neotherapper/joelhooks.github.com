<?xml version="1.0" encoding="utf-8"?>
<mx:Canvas 
    xmlns:mx="http://www.adobe.com/2006/mxml" 
    height="30"
    xmlns:pleatClasses="com.joelhooks.containers.pleatClasses.*"      
    verticalScrollPolicy="off" 
    horizontalScrollPolicy="off" 
    creationComplete="init()" 
    width="100%" 
    borderStyle="solid" 
    borderThickness="0" 
    cornerRadius="3">
     <mx:Script>
         <![CDATA[
             import com.joelhooks.containers.pleatClasses.controls.HelpControls;
             import com.joelhooks.containers.pleatClasses.PleatItem;
             import mx.effects.easing.Quadratic;
             import com.gskinner.motion.GTween;
             import com.joelhooks.containers.pleatClasses.PleatEvent;
             
             public var selectedIndex:int = 0;
             
             [Inspectable(arrayType="com.joelhooks.containers.pleatClasses.PleatItem")]
             [Bindable]
             private var _pleatItems:Array;
             public function get pleatItems():Array
             {
                 return this._pleatItems;
             }
             public function set pleatItems(value:Array):void
             {
                 this._pleatItems = value;
             }
             
             override protected function createChildren():void
             {
                 super.createChildren();
                 for each(var item:PleatItem in this._pleatItems)
                 {
                     item.addEventListener(PleatEvent.MOUSE_OUT_ITEM, handlePleatOut);
                     item.addEventListener(PleatEvent.MOUSE_OVER_ITEM, handlePleatOver);
                     item.addEventListener(PleatEvent.SELECT_ITEM, handleSelectItem); 
                     item.index = this.numChildren;
                     this.addChild(item);
                 }
             }
             
             
             
             private function init():void
             {
                 
                 this.addEventListener(Event.RESIZE, handleResize);
             }
             
             private function handleResize(event:Event):void
             {
                for(var i:int = 0;i<this.numChildren;i++)
                {
                    var child:PleatItem = this.getChildAt(i) as PleatItem;
                    if(child && child.selected)
                    {
                        child.dispatchEvent(new PleatEvent(PleatEvent.SELECT_ITEM));
                    }
                    if(child)
                    {
                        child.calculateOpenWidth();
                    }
                        
                }
             }
             
             private function handlePleatOver(event:PleatEvent):void
             {
                 if(!event.target.selected)
                 {    
                     event.target.calculateOpenWidth()
                     if(this.selectedIndex<event.target.index)
                         event.target.openLabel("left");
                     else
                         event.target.openLabel("right");
                 }
                    
                else
                    return;

                for(var i:int = 0;i<this.numChildren;i++)
                {
                    var child:PleatItem = this.getChildAt(i) as PleatItem;
                    child.calculateOpenWidth()
                    if(child && this.selectedIndex < child.index && event.target.index < child.index && this.selectPleatItemByIndex(this.selectedIndex)&& child != event.target)
                    {
                        //var selectedPleat:PleatItem = this.selectPleatItemByIndex(this.selectedIndex);
                        //child.reposition(selectedPleat.pleatLabel.width+selectedPleat.pleatContent.width+event.target.pleatLabel.width);
                    }
                    else if(child && this.selectedIndex == child.index &&  child.index>event.target.index&& this.selectPleatItemByIndex(this.selectedIndex)&& child != event.target)
                    {
                        child.reposition(event.target.pleatLabel.width);
                    }
                    else if(child && this.selectedIndex == child.index &&  child.index<event.target.index&& this.selectPleatItemByIndex(this.selectedIndex)&& child != event.target)
                    {
                        child.resizeContent(event.target.pleatLabel.width);
                    }
                    else if(child && child != event.target && child.index>event.target.index && !child.selected)
                    {
                        child.close(event.target.pleatLabel.width);
                    }
                    else if(child && child != event.target && child.index<event.target.index && event.target.selected)
                    {
                        child.close(event.target.pleatLabel.width+event.target.pleatContent.width);
                    }
                    else if(child && child!=event.target && !child.selected && child.index > this.selectedIndex && child.index<event.target.index)
                    {
                        child.reposition(this.selectPleatItemByIndex(this.selectedIndex).openWidth-30-event.target.pleatLabel.width);
                    }

                }
                
             }
             
             private function handlePleatOut(event:PleatEvent):void
             {
                 if(!event.target.selected)
                 {
                     event.target.calculateOpenWidth()
                     if(this.selectedIndex<event.target.index)
                         event.target.closeLabel("left");
                     else
                         event.target.closeLabel("right");
                 }
                    
                else
                    return;
                for(var i:int = 0;i<this.numChildren;i++)
                {
                    var child:PleatItem = this.getChildAt(i) as PleatItem;
                    child.calculateOpenWidth()
                    if(child && this.selectedIndex < child.index && this.selectPleatItemByIndex(this.selectedIndex)&& child != event.target)
                    {
                        var selectedPleat:PleatItem = this.selectPleatItemByIndex(this.selectedIndex);
                        child.reposition(selectedPleat.pleatLabel.width+selectedPleat.pleatContent.width);
                    }
                    else if(child && this.selectedIndex == child.index &&  child.index>event.target.index && this.selectPleatItemByIndex(this.selectedIndex)&& child != event.target)
                    {
                        child.reposition();
                    }
                    else if(child && this.selectedIndex == child.index &&  child.index<event.target.index&& this.selectPleatItemByIndex(this.selectedIndex)&& child != event.target)
                    {
                        child.resizeContent();
                    }
                    else if(child && child != event.target && !event.target.selected && !child.selected)
                    {
                        child.close();
                    }
                    else if(child && child != event.target && child.index>event.target.index && event.target.selected)
                    {
                        child.close(event.target.pleatLabel.width+event.target.pleatContent.width);
                    }
                    else if(child && child!=event.target && !child.selected && child.index > this.selectedIndex && child.index<event.target.index)
                    {
                        child.reposition(-60);
                    }

                }
             }
             
             private function handleSelectItem(event:PleatEvent):void
             {

                for(var i:int = 0;i<this.numChildren;i++)
                {
                    var child:PleatItem = this.getChildAt(i) as PleatItem;

                    if(child && child != event.target && event.target.index>child.index)
                    {
                        child.selected = false;
                        child.close()
                    }
                    else if(child && child != event.target && event.target.index<child.index)
                    {
                        child.selected = false;
                        child.close(event.target.pleatLabel.width+event.target.pleatContent.width);
                    }
                }
                
                 event.target.selected = true;
                 this.selectedIndex = event.target.index;
                 event.target.openContent()
            }
            
            private function selectPleatItemByIndex(index:int):PleatItem
            {
                for(var i:int = 0;i<this.numChildren;i++)
                {
                    var child:PleatItem = this.getChildAt(i) as PleatItem;
                    if(child && child.selected)
                        return child
                }
                return null;                    
            }
            
            public function getPleatItemCount():int
            {
                var count:int = 0;
                for(var i:int = 0;i<this.numChildren;i++)
                {
                    var child:PleatItem = this.getChildAt(i) as PleatItem;
                    if(child)
                        count++
                }                
                return count;
            }
         ]]>
     </mx:Script>    
</mx:Canvas>
