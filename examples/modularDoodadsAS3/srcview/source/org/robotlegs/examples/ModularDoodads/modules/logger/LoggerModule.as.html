package org.robotlegs.examples.ModularDoodads.modules.logger
{
    import flash.display.Sprite;
    import flash.text.AntiAliasType;
    import flash.text.Font;
    import flash.text.TextField;
    import flash.text.TextFieldAutoSize;
    import flash.text.TextFieldType;
    import flash.text.TextFormat;
    
    import org.robotlegs.core.IInjector;
    import org.robotlegs.utilities.modular.core.IModule;
    import org.robotlegs.utilities.modular.core.IModuleContext;
    
    public class LoggerModule extends Sprite implements IModule
    {
        [Embed("application/x-font", source="assets/AnonPro.ttf", fontFamily="Anon")]
        public var anon:Class;
        
        private var context:IModuleContext;
        private var text:TextField;
        
        public function LoggerModule()
        {
            drawBackground();
            initializeTextField();
        }
        
        [Inject]public function set parentInjector(value:IInjector):void
        {
            context = new LoggerModuleContext(this, value);
        }
        
        protected function drawBackground():void
        {
            graphics.beginFill(0x000000);
            graphics.drawRect(0,0,450,50);
            graphics.endFill();
        }
        
        protected function initializeTextField():void
        {
            var anonFont:Font = new anon;
            var tf:TextFormat = new TextFormat("Courier", 12, 0x00FF00);
            text = new TextField();
            text.type = TextFieldType.DYNAMIC;
            text.antiAliasType = AntiAliasType.ADVANCED;
            text.defaultTextFormat = tf;
            text.autoSize = TextFieldAutoSize.NONE;
            text.wordWrap = true;
            text.multiline = true;
            text.width = 440;
            text.height = 40;
            text.x = 5;
            text.y = 5;
            addChild(text);
        }
        public function addLoggingMessage(message:String):void
        {
            text.appendText(message + "\n");
            text.scrollV = text.numLines;
        }     
        public function dispose():void
        {
            context.dispose();
            text = null;
            context = null;
        }
    }
}
