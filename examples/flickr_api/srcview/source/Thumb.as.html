package
{
    import com.adobe.webapis.flickr.FlickrService;
    import com.adobe.webapis.flickr.Photo;
    import com.adobe.webapis.flickr.PhotoSize;
    import com.adobe.webapis.flickr.events.FlickrResultEvent;
    import com.gskinner.motion.GTween;
    
    import flash.display.Loader;
    import flash.display.Sprite;
    import flash.events.Event;
    import flash.events.IOErrorEvent;
    import flash.events.MouseEvent;
    import flash.events.ProgressEvent;
    import flash.net.URLRequest;
    import flash.net.navigateToURL;
    
    import mx.core.UIComponent;
    
    [Event(name="loaded",type="flash.events.Event")]
    
    public class Thumb extends UIComponent
    {
        private var loader:Loader = new Loader();
        private var _photo:Photo;
        public function get photo():Photo
        {
            return this._photo;
        }
        
        private var service:FlickrService;
        
        private var sizes:Array;
        private var coverSprite:Sprite;
        private var loadingBar:Sprite = new Sprite();
        private var loadingBarUnder:Sprite = new Sprite();
        
        public function Thumb(photo:Photo,service:FlickrService)
        {
            this.service = service;
            this._photo = photo
            this.width = 75;
            this.height = 75;
            this.init();
            super();
        }
        
        private function init():void
        {
            var request:URLRequest = new URLRequest(this.url);
            this.loader.load(request);
            
            this.loader.contentLoaderInfo.addEventListener(Event.COMPLETE, handleLoadComplete,false,0,true);
            this.loader.contentLoaderInfo.addEventListener(IOErrorEvent.IO_ERROR, handleIOEvent,false,0,true);
            this.loader.contentLoaderInfo.addEventListener(ProgressEvent.PROGRESS, handleProgress, false, 0, true);
            this.loader.alpha = 0;
            service.addEventListener(FlickrResultEvent.PHOTOS_GET_SIZES, getSizes_result)
            //service.photos.getSizes(this._photo.id);
            coverSprite = new Sprite()
            coverSprite.graphics.beginFill(0x000000,1)
            coverSprite.graphics.drawRect(0,0,75,75);
            this.coverSprite.alpha=.3
            
            this.buttonMode = true;
            this.mouseChildren = false;
            this.addEventListener(MouseEvent.MOUSE_OVER, handleMouseOver);
            this.addEventListener(MouseEvent.MOUSE_OUT, handleMouseOut);
            this.addEventListener(MouseEvent.CLICK, handleMouseClick);
            this.addChild(this.loadingBarUnder);
            this.loadingBarUnder.graphics.beginFill(0x333333,0);
            this.loadingBarUnder.graphics.drawRect(0,37,75,3);            
            this.addChild(this.loadingBar);
            this.loadingBar.graphics.beginFill(0x330000);
            this.loadingBar.graphics.drawRect(0,37,0,3);
            this.addChild(loader);
            this.addChild(coverSprite)
        }
        
        public function get url():String
        {
            return 'http://farm' + photo.farmId + '.static.flickr.com/' + photo.server + '/' + photo.id + '_' + photo.secret + '_s.jpg';
        }
        
        public function getLargestPhoto():PhotoSize
        {
            if(!this.sizes)
                return null;
            return this.sizes[this.sizes.length-1] as PhotoSize;
        }
        
        private function getSizes_result(event:FlickrResultEvent):void
        {
            this.sizes = event.data.photoSizes;
        }
        
        private function handleLoadComplete(event:Event):void
        {
            this.dispatchEvent(new Event("loaded"));
            var tween:GTween = new GTween(this.loader,1,{alpha:1});
        }
        
        private function handleIOEvent(event:IOErrorEvent):void
        {
        }
        
        private function handleMouseOver(event:MouseEvent):void
        {
            this.coverSprite.alpha = 0;
        }
        private function handleMouseOut(event:MouseEvent):void
        {
            this.coverSprite.alpha = .3;
        }
        private function handleMouseClick(event:MouseEvent):void
        {
            navigateToURL(new URLRequest("http://www.flickr.com/photos/"+this.photo.ownerId+"/"+this.photo.id));
        }
        
        private function handleProgress(event:ProgressEvent):void
        {
            this.loadingBar.graphics.clear();
            this.loadingBar.graphics.beginFill(0x330000);
            this.loadingBar.graphics.drawRect(0,37,(event.bytesLoaded/event.bytesTotal)*75,3);
            if(this.contains(this.loadingBar) && event.bytesLoaded==event.bytesTotal)
            {
                this.removeChild(this.loadingBar)
                this.removeChild(this.loadingBarUnder)
            }
                
        }
    }
}
